```typescript
/**
 * Format percentage values for display
 * @param value - Numeric percentage value (5.5 for 5.5%)
 * @param decimals - Number of decimal places (default: 1)
 * @returns Formatted percentage string with % suffix
 *
 * Examples:
 *   formatPercentage(5.5) → "5.5%"
 *   formatPercentage(3.0) → "3.0%"
 *   formatPercentage(10.25, 2) → "10.25%"
 *   formatPercentage(null) → "—"
 */
const formatPercentage = (value: number | null | undefined, decimals: number = 1): string => {
  if (value === null || value === undefined || isNaN(value)) return '—';
  return `${value.toFixed(decimals)}%`;
};
```

**Key Features:**
- Type-safe: Accepts `number | null | undefined`
- Null handling: Returns em dash '—' for missing values
- Decimal control: Configurable precision (default: 1 decimal)
- Pattern consistency: Matches formatNum() and formatCurrency()
- Fully documented: JSDoc with examples and test cases
